Design — Section 2 : Tokens de design

Tout ce que tu peux modifier dans tailwind.config.js et globals.css sans toucher aux composants.

globals.css — variables Light / Dark

app/globals.css
/* Light mode — depuis Figma "SDS Light" */
:root {
--color-bg: #ffffff;
--color-bg-secondary: #f5f5f5;
--color-bg-brand: #111111;
--color-text: #111111;
--color-text-secondary: #737373;
--color-border: rgba(12,12,13,0.1);
}

/* Dark mode — depuis Figma "SDS Dark" */
.dark {
--color-bg: #111111;
--color-bg-secondary: #1c1c1c;
--color-bg-brand: #f5f5f5;
--color-text: #f5f5f5;
--color-text-secondary: #a3a3a3;
--color-border: rgba(255,255,255,0.1);
}

tailwind.config.js — tout en un endroit

tailwind.config.js
theme: {
colors: {
bg: 'var(--color-bg)',
secondary: 'var(--color-bg-secondary)',
brand: 'var(--color-bg-brand)',
text: 'var(--color-text)',
muted: 'var(--color-text-secondary)',
border: 'var(--color-border)',
},
fontFamily: {
sans: ['Inter', 'sans-serif'], // Figma: Family Sans
serif: ['Noto Serif', 'serif'], // Figma: Family Serif
mono: ['Roboto Mono', 'monospace'],
},

page: '48px', heading: '24px',

},

},

}
